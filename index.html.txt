     * State

    const wheel = document.getElementById("wheel");
    const spinBtn = document.getElementById("spinBtn");
    const resetBtn = document.getElementById("resetBtn");

    const spinCountText = document.getElementById("spinCountText");

    const statusTitle = document.getElementById("statusTitle");

    let spins = 0;
    let angle = 0;
    let isSpinning = false;

    document.getElementById("year").textContent = new Date().getFullYear();

    function setStatus(kind, title, text){
      statusTitle.textContent = title;
      statusText.textContent = text;

      if(kind === "ready"){
        statusDot.style.background = "rgba(255,255,255,.35)";
      } else if(kind === "progress"){
        statusDot.style.background = "rgba(41,211,255,.95)";
      } else if(kind === "ok"){
        statusDot.style.background = "rgba(60,255,181,.95)";
      } else {
        statusDot.style.background = "rgba(255,77,109,.95)";
      }
    }

    function updateCenter(){
      const next = Math.min(spins + 1, REQUIRED_SPINS);
      if(spins < REQUIRED_SPINS){
        spinCountText.textContent = `Spin ${next} of ${REQUIRED_SPINS}`;
        hintText.textContent = spins === 0 ? "Tap Spin to begin" : "One more spin to check status";
      } else {
        spinCountText.textContent = "Completed";
        hintText.textContent = "You may proceed";
      }
    }

    // Simple celebratory sound (no external file)
    function playCelebration(){
      try{
        const ctx = new (window.AudioContext || window.webkitAudioContext)();
        const now = ctx.currentTime;

        const o1 = ctx.createOscillator();
        const g1 = ctx.createGain();
        o1.type = "sine";
        o1.frequency.setValueAtTime(660, now);
        o1.frequency.exponentialRampToValueAtTime(990, now + 0.18);
        g1.gain.setValueAtTime(0.0001, now);
        g1.gain.exponentialRampToValueAtTime(0.25, now + 0.02);
        g1.gain.exponentialRampToValueAtTime(0.0001, now + 0.28);
        o1.connect(g1).connect(ctx.destination);
        o1.start(now);
        o1.stop(now + 0.30);

        const o2 = ctx.createOscillator();
        const g2 = ctx.createGain();
        o2.type = "triangle";
        o2.frequency.setValueAtTime(440, now + 0.05);
        o2.frequency.exponentialRampToValueAtTime(740, now + 0.22);
        g2.gain.setValueAtTime(0.0001, now + 0.05);
        g2.gain.exponentialRampToValueAtTime(0.18, now + 0.07);
        g2.gain.exponentialRampToValueAtTime(0.0001, now + 0.32);
        o2.connect(g2).connect(ctx.destination);
        o2.start(now + 0.05);
        o2.stop(now + 0.34);
      } catch(e){
        // ignore if audio blocked
      }
    }

    function launchConfetti(){
      confetti.style.display = "block";
      confetti.innerHTML = "";
      const pieces = 28;
      for(let i=0;i<pieces;i++){
        const p = document.createElement("i");
        p.style.left = Math.random()*100 + "%";
        p.style.animationDelay = (Math.random()*0.15) + "s";
        p.style.transform = `translateY(0) rotate(${Math.random()*120}deg)`;
        // random bright-ish color via HSL
        const hue = Math.floor(Math.random()*360);
        p.style.background = `hsl(${hue} 95% 60%)`;
        p.style.width = (6 + Math.random()*8) + "px";
        p.style.height = (10 + Math.random()*10) + "px";
        confetti.appendChild(p);
      }
      setTimeout(()=>{ confetti.style.display="none"; }, 1400);
    }

    function buildRedirectUrl(){
      if(!PASS_THROUGH_QUERY) return CPA_URL;
      const q = window.location.search || "";
      if(!q) return CPA_URL;
      // append with proper separator
      const sep = CPA_URL.includes("?") ? "&" : "?";
      return CPA_URL + sep + q.replace(/^\?/, "");
    }

    function showCTA(){
      ctaBox.style.display = "block";
      launchConfetti();
      playCelebration();
      setStatus("ok", "Status: Opportunity unlocked", "Congratulations! You may be eligible to proceed.");
    }

    function doSpin(){
      if(isSpinning) return;
      if(spins >= REQUIRED_SPINS) return;

      isSpinning = true;
      spinBtn.disabled = true;
      resetBtn.disabled = true;

      setStatus("progress", "Status: Checking...", "Spinning to check reward status. Please wait.");

      // Random spin power; keep it non-deterministic but not implying a ‚Äúwin‚Äù
      const extraTurns = 5 + Math.floor(Math.random()*3); // 5-7 full turns
      const slice = 360 / 8;
      const randomOffset = Math.floor(Math.random()*8) * slice + Math.random() * (slice * 0.35);

      angle = angle + extraTurns*360 + randomOffset;

      wheel.style.transform = `rotate(${angle}deg)`;

      // when transition ends
      setTimeout(()=>{
        spins += 1;
        updateCenter();

        if(spins < REQUIRED_SPINS){
          setStatus("ready", "Status: One more step", "Spin again to complete the eligibility check.");
          spinBtn.disabled = false;
          resetBtn.disabled = false;
          isSpinning = false;
        } else {
          // Completion
          hintText.textContent = "Eligibility step unlocked";
          showCTA();
          resetBtn.disabled = false;
          isSpinning = false;
        }
      }, 4300);
    }

    function resetAll(){
      spins = 0;
      angle = 0;
      isSpinning = false;
      wheel.style.transition = "none";
      wheel.style.transform = "rotate(0deg)";
      // force reflow then restore transition
      void wheel.offsetWidth;
      wheel.style.transition = "transform 4.2s cubic-bezier(.12,.72,.10,1)";

      ctaBox.style.display = "none";
      spinBtn.disabled = false;
      resetBtn.disabled = true;

      setStatus("ready", "Status: Ready", "No claim of guaranteed winnings. This is an opportunity check.");
      updateCenter();
    }

    spinBtn.addEventListener("click", doSpin);
    resetBtn.addEventListener("click", resetAll);

    continueBtn.addEventListener("click", ()=>{
      // Safe redirect
      window.location.href = buildRedirectUrl();
    });

    updateCenter();
